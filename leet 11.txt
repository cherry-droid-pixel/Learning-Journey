class Solution {
    public int maxArea(int[] height) {
        int i=0;
        int j=height.length-1;
        int x=0;
        while(i<j){
            int h=Math.min(height[j],height[i]);
            x=Math.max(x,(h*(j-i)));
            if(height[i]<height[j]){
                i++;
            }else{
                j--;
            }
        }
        return x;
    }
}
